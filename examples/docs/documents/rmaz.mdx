# rmaz

Coordinate `requestMIDIAccess` calls across your app and/or libraries.

based on rafz: `requestAnimationFrame`

- raf => rma
- loop => change
- frame => access
- animation => midi

```js
import { rma } from 'rmaz'

// Schedule an update
rma(ts => {})

// Start an update loop
rma(ts => true)

// Cancel an update
rma.cancel(fn)

// Schedule a mutation
rma.write(() => {})

// Before any updates
rma.onStart(() => {})

// Before any mutations
rma.onAccess(() => {})

// After any mutations
rma.onFinish(() => {})

// Use a polyfill
rma.use(require('@essentials/rma').rma)
```


## Basic Examples

import { rma } from 'use-midi/src'
export let midi;

export function App () {
    React.useEffect(rma, [])
    return (
        <>
          <h4>inputs</h4>
          {Array.from(rma.inputs || []).map((v, i) =>
            <div key={i}>
              <h5>{v[0]}</h5>
              <div>name: {v[1].name}</div>
              <div>type: {v[1].type}</div>
            </div>
          )}
          <h4>outputs</h4>
          {Array.from(rma.outputs || []).map((v, i) =>
            <div key={i}>
              <h5>{v[0]}</h5>
              <div>name: {v[1].name}</div>
              <div>type: {v[1].type}</div>
            </div>
          )}
        </>
    )
}

<App />
