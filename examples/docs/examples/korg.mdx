export function PickColor (props) {
    const {children, ...other} = props
    const [color, set] = React.useState('black')
    const handleChange = e => set(e.target.value)
    return (
      <div {...other}>
        <input type="color" value={color} onChange={handleChange}/>
        {children(color)}
      </div>
    )
}
