function Tut (props) {
  const {note, other} = props
  const ref = React.useRef()
  const bind - useNote(state => {
    const {noteNum, velocity, send, args} = state
    if (note === noteNum)
      ref.current.active = velocity > 0
  }, {key: 0})
  return <button ref={ref} {...other} {...bind()}/>
}
